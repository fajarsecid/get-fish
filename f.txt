    local function isGV(obj)
        if not obj or not obj:IsDescendantOf(player.PlayerGui) then return false end
        local current = obj
        while current and current ~= game do
            if current:IsA("GuiObject") and not current.Visible then return false end
            if current:IsA("ScreenGui") and not current.Enabled then return false end 
            current = current.Parent
        end
        return true
    end

    local function BFT(x, y, state)
        pcall(function() VirtualInputManager:SendMouseButtonEvent(x, y, 0, state, game, 0) end)
        pcall(function() if state then VirtualUser:Button1Down(Vector2.new(x, y)) else VirtualUser:Button1Up(Vector2.new(x, y)) end end)
        
        if typeof(firesignal) == "function" then
            pcall(function()
                if state then
                    pressedElements = player.PlayerGui:GetGuiObjectsAtPosition(x, y)
                    for _, g in ipairs(pressedElements) do
                        pcall(function() firesignal(g.InputBegan, {UserInputType = Enum.UserInputType.Touch, UserInputState = Enum.UserInputState.Begin}) end)
                        pcall(function() firesignal(g.MouseButton1Down) end)
                    end
                else
                    for _, g in ipairs(pressedElements) do
                        pcall(function() firesignal(g.InputEnded, {UserInputType = Enum.UserInputType.Touch, UserInputState = Enum.UserInputState.End}) end)
                        pcall(function() firesignal(g.MouseButton1Up) end)
                        pcall(function() firesignal(g.MouseButton1Click) end)
                    end
                    pressedElements = {} 
                end
            end)
        end
    end

    local function ForceRelease()

    local function setLayarDitahan(state, isEmergency)
        local cooldown = isEmergency and emergencyCooldown or normalCooldown
        if isHolding ~= state and (os.clock() - lastSwitchTime) >= cooldown then
            isHolding = state; lastSwitchTime = os.clock()
            local cx = castMarker and math.floor(castMarker.AbsolutePosition.X + castMarker.AbsoluteSize.X / 2) or 0
            local cy = castMarker and math.floor(castMarker.AbsolutePosition.Y + castMarker.AbsoluteSize.Y / 2) or 0
            pcall(function() VirtualInputManager:SendMouseButtonEvent(cx, cy, 0, state, game, 1) end)
        end
    end

    local function cariMG_Cached(root)
        if not cachedRedBar or not cachedYellowZone or not cachedRedBar.Parent or not cachedYellowZone.Parent then
            local rd, yz, br, by = nil, nil, -1, -1
            for _, v in pairs(root:GetDescendants()) do
                if (v:IsA("Frame") or v:IsA("ImageLabel") or v:IsA("ImageButton")) and isGV(v) then
                    local c = v:IsA("Frame") and v.BackgroundColor3 or v.ImageColor3
                    local sx, sy = v.AbsoluteSize.X, v.AbsoluteSize.Y
                    if c.R > 0.75 and c.G < 0.35 and c.B < 0.35 then
                        local r = math.max(sx, sy) / math.max(1, math.min(sx, sy))
                        local s = math.min(sx, sy)
                        if r <= 2 and s > br then br = s; rd = v end
                    end
                    if c.R > 0.75 and c.G > 0.75 and c.B < 0.35 and sx > sy and sx > by then by = sx; yz = v end
                end
            end
            cachedRedBar = rd
            cachedYellowZone = yz
        end
        return cachedRedBar, cachedYellowZone
    end

    local function kontrolMinigame(rd, yz)
        local rawX = getCX(rd)
        local zL, zR = getBX(yz)
        local zW = math.max(1, zR - zL)
        local trim = math.max(2, zW * 0.05) 
        zL = zL + trim; zR = zR - trim; zW = math.max(1, zR - zL)
        local zCenter = (zL + zR) / 2; local halfZone = zW / 2

        if not smoothedPos then smoothedPos = rawX; lastPos = rawX; smoothedVel = 0
        else smoothedPos = smoothedPos + (rawX - smoothedPos) * posAlpha end

        local rawVel = smoothedPos - lastPos
        smoothedVel = smoothedVel + (rawVel - smoothedVel) * velAlpha
        lastPos = smoothedPos

        local predictedX = smoothedPos + smoothedVel * lookAhead
        local err = predictedX - zCenter
        local normErr = err / halfZone
        local isEmergency = math.abs(normErr) > emergencyFrac or predictedX < zL or predictedX > zR

        if normErr > deadzoneFrac then setLayarDitahan(true, isEmergency)
        elseif normErr < -deadzoneFrac then setLayarDitahan(false, isEmergency) end

        local errPct = math.floor(normErr * 100)
        local velPx = math.floor(smoothedVel * 100) / 100
        dbgLabelFish.Text = "err:" .. errPct .. "% vel:" .. string.format("%.1f", velPx) .. " hold:" .. tostring(isHolding)
    end

    local function RefreshPlayerList()
        for _, child in pairs(playerScroll:GetChildren()) do 
            if child:IsA("TextButton") then child:Destroy() end 
        end
        local ySize = 0
        for _, p in pairs(Players:GetPlayers()) do
            if p ~= player then
                local btn = Instance.new("TextButton", playerScroll)
                btn.Size = UDim2.new(0.9, 0, 0, 30) 
                btn.BackgroundColor3 = Color3.fromRGB(40, 30, 50) 
                btn.TextColor3 = Color3.fromRGB(200, 150, 255) 
                btn.Font = Enum.Font.GothamSemibold 
                btn.TextSize = 12 
                btn.Text = p.Name 
                btn.ZIndex = 102
                Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 5)
                
                btn.MouseButton1Click:Connect(function()
                    selectedPlayer = p
                    btnBukaMenuPlayer.Text = "Target: " .. (string.len(p.Name) > 10 and string.sub(p.Name, 1, 9)..".." or p.Name)
                    playerListFrame.Visible = false

                        local thumbSize = Enum.ThumbnailSize.Size420x420

                        previewImage.Image = content
                    end)
                end)
                ySize = ySize + 35
            end
        end
        playerScroll.CanvasSize = UDim2.new(0, 0, 0, ySize)
    end

BtnVerify.MouseButton1Click:Connect(function()
    KeyStatus.TextColor3 = Color3.fromRGB(255, 180, 0)
    KeyStatus.Text = "Memeriksa Key..."
    
    local inputKey = KeyBox.Text
    if inputKey == "" then
        KeyStatus.TextColor3 = Color3.fromRGB(200, 50, 50)
        KeyStatus.Text = "Key tidak boleh kosong!"
        return
    end

    local success, result = pcall(function() 
        return game:HttpGet(RAW_KEY_URL) 
    end)
    
    if success then
        local onlineKey = string.gsub(result, "^%s*(.-)%s*$", "%1")
        if inputKey == onlineKey then
            KeyStatus.TextColor3 = Color3.fromRGB(50, 200, 50)
            KeyStatus.Text = "Key Benar! Memuat Script..."
            task.wait(1)
            LoadMainScripts()
        else
            KeyStatus.TextColor3 = Color3.fromRGB(200, 50, 50)
            KeyStatus.Text = "Key Salah! Silakan coba lagi."
        end
    else
        KeyStatus.TextColor3 = Color3.fromRGB(200, 50, 50)
        KeyStatus.Text = "Gagal terhubung ke server (Error HttpGet)."
    end
end)
