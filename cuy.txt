function AntiDetect.RandomChildName(parent)
    if parent then
        for _, child in pairs(parent:GetChildren()) do
            if child:IsA("ScreenGui") or child:IsA("Frame") or child:IsA("TextButton")
            or child:IsA("TextLabel") or child:IsA("TextBox") or child:IsA("ScrollingFrame")

            end

        end
    end
end

function AntiDetect.AntiKick()
    pcall(function()
        if hookmetamethod and getnamecallmethod then
            local oldNamecall
            oldNamecall = hookmetamethod(game, "__namecall", function(self, ...)
                local method = getnamecallmethod()
                local args = {...}
                if method == "Kick" then return nil end
                if method == "TeleportToPlaceInstance" or method == "Teleport" then
                    local info = tostring(args[1] or "")
                    if string.find(string.lower(info), "ban") or string.find(string.lower(info), "kick") then return nil end
                end
                return oldNamecall(self, ...)
            end)
        end
    end)
end

function AntiDetect.HideRemoteCalls()
    pcall(function()

                return oldFireServer(self, ...)
            end)
        end
    end)
end

end

function AntiDetect.BlockDetectionRemotes()
    pcall(function()
        local suspiciousNames = {"report", "anticheat", "detect", "cheat", "exploit", "hack", "ban", "kick", "flag", "security", "validate", "check", "monitor", "watchdog", "guard", "verification", "integrity", "suspicion", "violation"}
        if hookfunction then
            local oldFire
            oldFire = hookfunction(Instance.new("RemoteEvent").FireServer, function(self, ...)
                local remoteName = string.lower(self.Name)
                for _, keyword in pairs(suspiciousNames) do if string.find(remoteName, keyword) then return nil end end
                return oldFire(self, ...)
            end)
            local oldInvoke
            oldInvoke = hookfunction(Instance.new("RemoteFunction").InvokeServer, function(self, ...)
                local remoteName = string.lower(self.Name)
                for _, keyword in pairs(suspiciousNames) do if string.find(remoteName, keyword) then return nil end end
                return oldInvoke(self, ...)
            end)
        end
    end)
end

function AntiDetect.CleanLogs()
    pcall(function()
        local realPrint = print
        local realWarn = warn
        local filterWords = {"cheat", "hack", "exploit", "fajar", "delta", "auto fish", "esp", "noclip"}
        print = function(...)
            local msg = table.concat({...}, " "):lower()
            for _, word in pairs(filterWords) do if string.find(msg, word) then return end end
            return realPrint(...)
        end
        warn = function(...)
            local msg = table.concat({...}, " "):lower()
            for _, word in pairs(filterWords) do if string.find(msg, word) then return end end
            return realWarn(...)
        end
    end)
end

    end)
end

local function safeTeleport(targetCFrame)
    local char = player.Character
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    if hrp then
        local distance = (hrp.Position - targetCFrame.Position).Magnitude
        local speed = math.clamp(distance / 500, 0.5, 1.5)
        local tweenInfo = TweenInfo.new(speed, Enum.EasingStyle.Linear)
        local tween = TweenService:Create(hrp, tweenInfo, {CFrame = targetCFrame})
        local bv = Instance.new("BodyVelocity", hrp)
        bv.Velocity = Vector3.new(0, 0, 0)
        bv.MaxForce = Vector3.new(100000, 100000, 100000)
        tween:Play()
        tween.Completed:Wait()
        bv:Destroy()
    end
end

local fpsConnection = nil
local origShadows = Lighting.GlobalShadows
local origFog = Lighting.FogEnd

btnFPS.MouseButton1Click:Connect(function()
    isFPS = not isFPS
    toggleButtonVisual(btnFPS, strokeFPS, isFPS, "FPS Unlocker Max")
    local terrain = workspace:FindFirstChildOfClass("Terrain")
    if isFPS then
        pcall(function() setfpscap(0) end)
        origShadows = Lighting.GlobalShadows
        origFog = Lighting.FogEnd
        Lighting.GlobalShadows = false
        Lighting.FogEnd = 9e9
        if terrain then
            terrain.WaterWaveSize = 0
            terrain.WaterWaveSpeed = 0
            terrain.WaterReflectance = 0
        end
        local function optimizeObject(v)
            pcall(function()
                if v:IsA("BasePart") and v.Name ~= "Terrain" then
                    v.Material = Enum.Material.SmoothPlastic
                    v.CastShadow = false
                    v.Reflectance = 0
                elseif v:IsA("Decal") or v:IsA("Texture") then
                    v.Transparency = 1
                elseif v:IsA("Light") then
                    v.Enabled = false
                elseif v:IsA("Highlight") then
                    v.Enabled = false
                elseif v:IsA("PostEffect") then
                    v.Enabled = false
                elseif v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Beam") or v:IsA("Sparkles") or v:IsA("Fire") or v:IsA("Smoke") then
                    v.Enabled = false
                end
            end)
        end
        for _, v in pairs(workspace:GetDescendants()) do optimizeObject(v) end
        for _, v in pairs(Lighting:GetChildren()) do optimizeObject(v) end
        if player.Character then
            for _, v in pairs(player.Character:GetDescendants()) do optimizeObject(v) end
        end
        fpsConnection = workspace.DescendantAdded:Connect(function(v) optimizeObject(v) end)
    else
        pcall(function() setfpscap(60) end)
        Lighting.GlobalShadows = origShadows
        Lighting.FogEnd = origFog
        if terrain then
            terrain.WaterWaveSize = 0.15
            terrain.WaterWaveSpeed = 10
            terrain.WaterReflectance = 1
        end
        if fpsConnection then fpsConnection:Disconnect() fpsConnection = nil end
        pcall(function()
            for _, v in pairs(Lighting:GetChildren()) do
                if v:IsA("PostEffect") then v.Enabled = true end
            end
            for _, v in pairs(workspace:GetDescendants()) do
                if v:IsA("Light") or v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Beam") or v:IsA("Highlight") then
                    v.Enabled = true
                elseif (v:IsA("Decal") or v:IsA("Texture")) then
                    v.Transparency = 0
                end
            end
        end)
    end
end)
